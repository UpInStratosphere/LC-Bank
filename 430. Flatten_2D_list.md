- recursive Helper function connects the lower level connected list to the current level

```cpp
Node* flatten(Node* head) {
    Node* start = head;
    Node* upper = NULL;
    connect(start, upper);
    return head;
}

void connect(Node* curr, Node* upper){ 
    while(curr){
        // connect the lower level nodes and return to curr level
        if(curr->child){ 
            Node* todo = curr->next; 
            
            curr->next = curr->child; 
            curr->child->prev = curr;
            curr->child = NULL;       
            connect(curr->next, todo); 

            curr = todo;
            continue;
        }

        //connect curr level to upper level 
        if(!curr->next && upper){
            curr->next = upper;
            upper->prev = curr;
            return;
        } 
        
        //keep traversing the curr level
        curr = curr->next; 
    }  
}

//iterative solution is much more understandable. Each iteration determine the correct next pointer and move to the next node
Node* flatten(Node* head) {
    if (!head) return NULL;
    stack<Node*>stk;
    Node *curr = head;
    while (curr){
        if (curr->child){
            if (curr->next) 
                stk.push(curr->next);
            curr->next= curr->child;
            curr->next->prev = curr;
            curr->child = NULL;
        }
        else if (!curr->next && !stk.empty()){
            curr->next = stk.top();
            stk.pop();
            curr->next->prev = curr;
        }
        curr = curr->next;
    }
    return head;
}
 ```
