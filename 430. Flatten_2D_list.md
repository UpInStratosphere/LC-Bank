- recursive Helper function connects the lower level connected list to the current level

```cpp
 Node* flatten(Node* head) {
    if (!head) return NULL;
    helper(head); 
    return head;
}

//function returns the last node of the curr flattened level
Node* helper(Node *curr){ 
    while (curr){
        if (curr->child){ //there is a subproblem
            Node *resume = curr->next;

            curr->next = curr->child;
            curr->next->prev = curr;
            curr->child = NULL;

            //subproblems parameter : the head node of the lower list 
            curr = helper(curr->next); 
            if (resume){
                curr->next = resume;
                resume->prev = curr;
            }
        }
        if (!curr->next)
            return curr;
        else
            curr = curr->next;
    }
    return NULL;
}
```
