- outer brackets cannot be unpacked UNTIL its all the included inner brackets are have their data unpacked.

```cpp
string decodeString(string s) {
    if (s.empty()) return "";
    int curr_num = 0;
    string curr_str = "";
    stack<int>nums;
    stack<string>strs;
    int n = s.size();
    for (char c : s){
        if (isdigit(c)
            curr_num = curr_num * 10 + (c-'0');
        else if (c == '['){
            nums.push(curr_num);
            strs.push(curr_str);
            curr_str = "";
            curr_num = 0;
        }
        else if (isalpha(c))
            curr_str.push_back(c);
        else{
            string temp_str = curr_str;
            int temp_num = nums.top();
            nums.pop();
            while (temp_num > 1){
                curr_str += temp_str;
                temp_num--;
            }
            curr_str = strs.top() + curr_str;
            strs.pop();
        }
    }
    return curr_str;
}
```
