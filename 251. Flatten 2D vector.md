class Vector2D {
    vector<vector<int>>::iterator currentRow, endRow;
    int pos = 0;

public:
    Vector2D(vector<vector<int>>& v) {
        currentRow = v.begin();
        endRow = v.end();
    }
    
    int next() { //this assumes this function is always valid, will not be out of the 2Dvector boundaries. Else, we need to determine a sentinal integer value to be returned whenever we are out of the boundary. 
        hasNext(); //only used to increment the index position, not using the returned value of the function
        pos++;
        return (*currentRow)[pos]; 
    }
        /* if (hasNext()){

    bool hasNext() {
        //to skip any potential empty vectors
        //when the current row has reached the last element, move to the next non-empty vector
        while (currentRow != endRow && pos == (*currentRow).size()) 
            currentRow++, pos = 0;
        return currentRow != endRow;  
    }
};
