    if (nums.empty()) return 0;
    int res = 0, n = nums.size();
    unordered_set<int>unique(nums.begin(), nums.end());

        while (unique.find(prev) != unique.end()){
            curr += 1;
            unique.erase(prev);
            prev = prev-1;
        }
        while (unique.find(next) != unique.end()){
            curr += 1;
            unique.erase(next);
            next = next+1;
        }
        
        res = max(res, curr);
    }
    return res;
}
